库打桩（lib interposition）

库打桩技术的使用场景：可以通过截获共享库的调用，执行自己的代码，通常使用该技术调试代码。库打桩实现有以下三种:编译时打桩，链接时打桩， 运行时打桩。

编译时（compile Time）

  • 文件排布
├── main
├── main.c
├── malloc.h
├── my_malloc.c
└── my_malloc.o

// main.c
#include "stdio.h"
#include <malloc.h>

int main(){

	int* test = (int *) malloc(32);
	printf("addrs is %p\n", test);
	free(test);
	return 0;
}

// my_malloc.c
#ifdef COMPILETIME
#include "stdlib.h"
#include "stdio.h"
void *my_malloc(size_t size) {
	printf("my malloc\n");
	void *ptr = malloc(size);
	printf("alloc %zu size of memery and addrs is %p\n", size, ptr);
	return ptr;
}

void my_free(void *ptr){
	printf("my free\n");
	printf("free mempry at %p\n", ptr);
	free(ptr);
}
#endif

// malloc.h
#define malloc(size) my_malloc(size)
#define free(ptr) my_free(ptr)

void *my_malloc(size_t size);
void my_free(void *ptr);
  • compile command
gcc -D COMPILETIME -c my_malloc.c 
gcc -Wall -I . -o main main.c my_malloc.o

链接时（LinkerTime）

通过-Wl将参数传输给linker，实现打桩

#include "stdio.h"
// 需要添加前置声明，否则会core
void *__real_malloc(size_t size);
void __real_free(void *ptr);

void *__wrap_malloc(size_t size) {
    void *ptr = __real_malloc(size);
    fprintf(stderr, "alloc %zu bits at %p\n",size, ptr);
    return ptr;
}

void __wrap_free(void *ptr) {
    fprintf(stderr, "dynamic free %p\n", ptr);
    if (ptr == NULL) {
        return;
    }
    __real_free(ptr);
    fprintf(stderr, "realse memery buffer at %p\n", ptr);
}

// main.cpp
#include "stdio.h"
#include "stdlib.h"

int main(){

    int *ptr = (int *)malloc(32);
    free(ptr);
    return 0;
}
  • 编译指令, 通过Wl实现库打桩
gcc -c my_malloc.c
gcc -Wl,--wrap,malloc -Wl,--wrap,free -o main main.c my_malloc.o

运行时（load/RunTime ）

需要注意的是printf会默认调用malloc，因此在my_malloc.c中malloc函数中需要调用fpirnf输出，否则会造成递归调用导致程序崩溃。

// my_malloc.c
#define _GNU_SOURCE
#include "stdio.h"
#include "stdlib.h"
#include "dlfcn.h"

void *malloc(size_t size) {
    void *(*mallocp)(size_t size);
    char *error;
    mallocp = dlsym(RTLD_NEXT, "malloc");
    if ((error = dlerror()) != NULL) {
        fputs(error, stderr);
        exit(1);
    }
    void *ptr = mallocp(size);
    fprintf(stderr, "malloc %zu bits memory at %p\n", size, ptr);
    return ptr;
}

void free(void *ptr) {
    void (*freep)(void *ptr);
    char *error;
    freep = dlsym(RTLD_NEXT, "free");
    if ((error = dlerror()) != NULL) {
        fputs(error, stderr);
        exit(1);
    }
    freep(ptr);
    fprintf(stderr, "free %p\n", ptr);
}

// main.c
#include "stdlib.h"
#include "stdio.h"

int main(){
    fprintf(stderr, "start!\n");
    int * ptr = (int *)malloc(32);
    free(ptr);
    fprintf(stderr, "end!\n");
    return 0;
}
  • 编译
 gcc -shared -fpic -o my_malloc.so my_malloc.c -ldl
 gcc -o main main.c
